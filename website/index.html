Quick Start

variables.fenics 
// Constant variables (immutable)
const name : "Alice"
const age : 25

// Mutable variables
counter : 0
counter +: 1  // Add-assign

// String interpolation
print("#{name} is #{age} years old")
functions.fenics 
// Function definition
fn add(a, b):
    return a + b

// Arrow functions
const multiply : fn(x, y) : x * y

// Using functions
const result : add(5, 3)
print("Result: #{result}")
loops.fenics 
// For loop
for i in 0..5:
    print("Iteration #{i}")

// While loop
count : 0
while count < 3:
    print(count)
    count +: 1
arrays.fenics 
// Array creation
const numbers : [1, 2, 3, 4, 5]
const names : ["Alice", "Bob", "Charlie"]

// Array operations
print("Length: #{len(numbers)}")
print("First: #{numbers[0]}")
